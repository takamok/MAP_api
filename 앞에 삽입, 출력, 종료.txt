typedef struct cell
{
	int element;
	int num;
	struct cell *next;

}node;
node *head;

void insert1(int val)
{
	node *tmp;
	tmp = (node *)malloc(sizeof(node));
	tmp->element = val;
	tmp->next = head;
	head = tmp;

}


void print_list()
{
	node *tmp = head;

	while (tmp != NULL)
	{
		printf("%5d", tmp->element);
		tmp = tmp->next;
	}



}
